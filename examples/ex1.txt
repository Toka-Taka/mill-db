CREATE TABLE Data (
	id int,
	value float,
	weight double
);

CREATE INDEX Data_PK on Data (id);

CREATE PROCEDURE writeproc_Data (@param1 int in, @param2 float in, @param3 double in)
BEGIN
	INSERT TABLE Data VALUES (@param1, @param2, @param3);
END;

CREATE PROCEDURE readproc_Data (@param1 int in, @param2 int out)
BEGIN
	SELECT value SET @param2 FROM Data WHERE id = @param1;
END;
